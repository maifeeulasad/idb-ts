UserEntity.ts 
import { Entity, PrimaryKey, Property } from 'idb-ts';

@Entity('users')
export class User {
  @PrimaryKey({ autoIncrement: true })
  id: number;

  @Property({ unique: true })
  email: string;

  @Property()
  username: string;
}

// Usage
const db = await IDB.build({ 
  entities: [User] 
});
